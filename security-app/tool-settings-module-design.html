5. Service 层设计

工具设置模块需要 两个 Service：一个负责用户认证，一个负责安全存储。

5.1 AuthService.ets（用户认证）

import { userAuth } from '@kit.UserAuthenticationKit'
import { hilog } from '@kit.PerformanceAnalysisKit'
import { AuthMethod } from '../models/DataModels'

const TAG = 'AuthService'
const DOMAIN = 0x0004

export class AuthService {

  // 检查认证方式是否可用
  static isAuthAvailable(authType: userAuth.UserAuthType): boolean {
    try {
      userAuth.getAvailableStatus(authType, userAuth.AuthTrustLevel.ATL1)
      return true
    } catch (err) {
      hilog.warn(DOMAIN, TAG, `Auth type ${authType} not available: ${JSON.stringify(err)}`)
      return false
    }
  }

  // 执行用户认证
  static async authenticate(authMethod: AuthMethod, challenge?: Uint8Array): Promise<boolean> {
    try {
      // 转换认证方式为 userAuth 类型数组
      let authTypes: userAuth.UserAuthType[] = AuthService.convertAuthMethod(authMethod)

      // 构建认证参数
      let authParam: userAuth.AuthParam = {
        challenge: challenge ?? new Uint8Array([0]),
        authType: authTypes,
        authTrustLevel: userAuth.AuthTrustLevel.ATL1
      }

      // 构建认证界面参数
      let widgetParam: userAuth.WidgetParam = {
        title: '身份验证',
        navigationButtonText: '取消'
      }

      // 获取认证实例
      let userAuthInstance = userAuth.getUserAuthInstance(authParam, widgetParam)

      // 执行认证（Promise 封装）
      return new Promise((resolve) => {
        userAuthInstance.on('result', {
          onResult(result: userAuth.UserAuthResult) {
            hilog.info(DOMAIN, TAG, `Auth result: ${result.result}`)
            resolve(result.result === userAuth.UserAuthResultCode.SUCCESS)
          }
        })
        userAuthInstance.start()
      })
    } catch (err) {
      hilog.error(DOMAIN, TAG, `authenticate failed: ${JSON.stringify(err)}`)
      return false
    }
  }

  // AuthMethod 转换为 userAuth.UserAuthType 数组
  private static convertAuthMethod(method: AuthMethod): userAuth.UserAuthType[] {
    switch (method) {
      case AuthMethod.PIN:
        return [userAuth.UserAuthType.PIN]
      case AuthMethod.FINGERPRINT:
        return [userAuth.UserAuthType.FINGERPRINT]
      case AuthMethod.FACE:
        return [userAuth.UserAuthType.FACE]
      case AuthMethod.PIN_FINGERPRINT:
        return [userAuth.UserAuthType.PIN, userAuth.UserAuthType.FINGERPRINT]
      default:
        return [userAuth.UserAuthType.PIN]
    }
  }
}

5.2 SecureStorageService.ets（安全存储）

import { asset } from '@kit.AssetStoreKit'
import { util } from '@kit.ArkTS'
import { hilog } from '@kit.PerformanceAnalysisKit'
import { ASSET_KEY_TOOL_PASSWORD } from '../models/DataModels'

const TAG = 'SecureStorageService'
const DOMAIN = 0x0005

export class SecureStorageService {

  // 保存工具密码
  static async savePassword(password: string): Promise<boolean> {
    try {
      let encoder = new util.TextEncoder()
      let passwordData = encoder.encodeInto(password)
      let aliasData = encoder.encodeInto(ASSET_KEY_TOOL_PASSWORD)

      // 先尝试查询是否已存在
      let exists = await SecureStorageService.hasPassword()

      if (exists) {
        // 更新已有密码
        let query: asset.AssetMap = new Map()
        query.set(asset.Tag.ALIAS, aliasData)

        let attrsToUpdate: asset.AssetMap = new Map()
        attrsToUpdate.set(asset.Tag.SECRET, passwordData)

        await asset.update(query, attrsToUpdate)
      } else {
        // 新增密码
        let attrs: asset.AssetMap = new Map()
        attrs.set(asset.Tag.ALIAS, aliasData)
        attrs.set(asset.Tag.SECRET, passwordData)
        attrs.set(asset.Tag.ACCESSIBILITY, asset.Accessibility.DEVICE_UNLOCKED)
        attrs.set(asset.Tag.AUTH_TYPE, asset.AuthType.NONE)

        await asset.add(attrs)
      }

      hilog.info(DOMAIN, TAG, 'Password saved successfully')
      return true
    } catch (err) {
      hilog.error(DOMAIN, TAG, `savePassword failed: ${JSON.stringify(err)}`)
      return false
    }
  }

  // 验证密码
  static async verifyPassword(inputPassword: string): Promise<boolean> {
    try {
      let encoder = new util.TextEncoder()
      let decoder = new util.TextDecoder()
      let aliasData = encoder.encodeInto(ASSET_KEY_TOOL_PASSWORD)

      let query: asset.AssetMap = new Map()
      query.set(asset.Tag.ALIAS, aliasData)
      query.set(asset.Tag.RETURN_TYPE, asset.ReturnType.ALL)

      let result = await asset.query(query)
      if (result.length === 0) {
        return false
      }

      let storedData = result[0].get(asset.Tag.SECRET) as Uint8Array
      let storedPassword = decoder.decodeToString(storedData)

      return storedPassword === inputPassword
    } catch (err) {
      hilog.error(DOMAIN, TAG, `verifyPassword failed: ${JSON.stringify(err)}`)
      return false
    }
  }

  // 检查密码是否已设置
  static async hasPassword(): Promise<boolean> {
    try {
      let encoder = new util.TextEncoder()
      let aliasData = encoder.encodeInto(ASSET_KEY_TOOL_PASSWORD)

      let query: asset.AssetMap = new Map()
      query.set(asset.Tag.ALIAS, aliasData)
      query.set(asset.Tag.RETURN_TYPE, asset.ReturnType.ATTRIBUTES)

      let result = await asset.query(query)
      return result.length > 0
    } catch (err) {
      return false
    }
  }

  // 删除密码
  static async removePassword(): Promise<boolean> {
    try {
      let encoder = new util.TextEncoder()
      let aliasData = encoder.encodeInto(ASSET_KEY_TOOL_PASSWORD)

      let query: asset.AssetMap = new Map()
      query.set(asset.Tag.ALIAS, aliasData)

      await asset.remove(query)
      hilog.info(DOMAIN, TAG, 'Password removed successfully')
      return true
    } catch (err) {
      hilog.error(DOMAIN, TAG, `removePassword failed: ${JSON.stringify(err)}`)
      return false
    }
  }
}

@Component
export struct ToolSettingsPage {
  @StorageLink(STORAGE_KEY_THEME) currentTheme: ThemeMode = ThemeMode.LIGHT

  // ========== 密码可见性状态 ==========

  @State showNewPwd: boolean = false
  @State showConfirmPwd: boolean = false

  // Select 选项
  private authMethodOptions: SelectOption[] = AUTH_METHOD_OPTIONS.map(
    (opt) => ({ value: opt.label })
  )

  aboutToAppear() { this.loadSettings() }

  build() {
    Column() {
      // 页面标题

7.4 保存逻辑

// 保存设置
async saveSettings() {
  // 1. 保存启动校验配置到 Preferences
  let prefs = await preferences.getPreferences(getContext(this), 'tool_settings')
  await prefs.put(PREF_KEY_STARTUP_AUTH, this.startupAuthEnabled)
  await prefs.put(PREF_KEY_AUTH_METHOD, AUTH_METHOD_OPTIONS[this.authMethodIndex].value)
  await prefs.flush()

  // 2. 如果有密码变更，验证并保存
  if (this.newPassword.length > 0) {
    // 验证新密码和确认密码一致
    if (this.newPassword !== this.confirmPassword) {
      promptAction.showToast({ message: '两次输入的密码不一致' })
      return
    }

    // 如果已有密码，先验证当前密码
    if (this.hasExistingPassword) {
      let verified = await SecureStorageService.verifyPassword(this.currentPassword)
      if (!verified) {
        promptAction.showToast({ message: '当前密码错误' })
        return
      }
    }

    // 保存新密码
    let saved = await SecureStorageService.savePassword(this.newPassword)
    if (!saved) {
      promptAction.showToast({ message: '密码保存失败' })
      return
    }
  }

  promptAction.showToast({ message: '设置已保存' })
}

1. Asset Store Kit 的所有 API 都是异步的，需要使用 async/await
2. 密码输入框需要支持明文/密文切换，提升用户体验
3. 首次使用时没有"当前密码"，需要判断 hasExistingPassword 状态
4. userAuth 调用系统认证界面后，结果通过回调返回，需要封装为 Promise
5. 认证方式的可用性检查应在页面加载时执行，禁用不支持的选项

数据模型

// 主题模式（实际显示）
enum ThemeMode {
  LIGHT = 'light',
  DARK = 'dark'
}

// 用户偏好（选择方式）
enum ThemePreference {
  SYSTEM = 'system',  // 跟随系统
  MANUAL = 'manual'   // 手动选择
}

// AppStorage 存储
AppStorage.setOrCreate<ThemeMode>('currentTheme', theme)      // 当前显示
AppStorage.setOrCreate<ThemePreference>('themePreference', pref) // 选择方式

// Preferences 持久化
await preferences.put('theme_mode', theme)
await preferences.put('theme_preference', pref)

推荐方案

推荐采用方案 B

选中"手动选择"后展开 Toggle 子选项，原因：

  • 层级清晰：先选择模式（跟随系统 vs 手动），再选择具体主题
  • 操作直观：Toggle 切换符合用户对"二选一"的心理预期
  • 空间利用：仅在手动模式下展示 Toggle，避免界面拥挤
  • 扩展友好：未来如需增加"自动"等模式，无需改变整体布局

ArkTS 组件实现

@Component
struct ThemeSettings {
  @StorageLink('currentTheme') currentTheme: ThemeMode = ThemeMode.LIGHT
  @StorageLink('themePreference') preference: ThemePreference = ThemePreference.SYSTEM

  build() {
    Column({ space: 12 }) {
      Text('外观模式').fontSize(14).fontColor('#999')
      
      Row({ space: 12 }) {
        // 跟随系统按钮
        Button('跟随系统')
          .onClick(() => ThemeManager.getInstance().setPreference(ThemePreference.SYSTEM))
          .backgroundColor(this.preference === ThemePreference.SYSTEM ? '#E3F2FD' : '#FFF')
        
        // 手动选择按钮
        Button('手动选择')
          .onClick(() => ThemeManager.getInstance().setPreference(ThemePreference.MANUAL))
          .backgroundColor(this.preference === ThemePreference.MANUAL ? '#E3F2FD' : '#FFF')
      }
      
      // 手动模式下的 Toggle
      if (this.preference === ThemePreference.MANUAL) {
        Row() {
          Toggle({ type: ToggleType.Button, isOn: this.currentTheme === ThemeMode.LIGHT })
            .onChange((isOn) => {
              ThemeManager.getInstance().setTheme(isOn ? ThemeMode.LIGHT : ThemeMode.DARK)
            })
          Text(this.currentTheme === ThemeMode.LIGHT ? '浅色' : '深色')
        }
      }
    }
  }
}

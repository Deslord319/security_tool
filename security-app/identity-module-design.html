关键设计点：UI 原型将口令复杂度拆分为 5 个独立配置项（长度、大写、小写、数字、特殊字符），但 PasswordPolicy API 只接收一个 complexityRegex 正则字符串。因此 Service 层需要一个正则生成器函数，根据用户选择的配置项动态拼装正则表达式。

5. Service 层设计

参照 FirewallService.ets / PeripheralService.ets 的静态类 + try-catch + hilog 模式，IdentityService.ets 需要封装 两组 API 以及一个核心的正则生成器。

5.1 正则生成器（核心逻辑）

/**
 * 根据 UI 配置项动态生成密码复杂度正则表达式
 * 示例输出: ^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)(?=.*[!@#$%^&*])[a-zA-Z\d!@#$%^&*]{8,}$
 */
static buildComplexityRegex(config: PasswordComplexityConfig): string {
  let lookaheads: string[] = []
  let charSet = ''

  if (config.requireLowercase) {
    lookaheads.push('(?=.*[a-z])')
    charSet += 'a-z'
  }
  if (config.requireUppercase) {
    lookaheads.push('(?=.*[A-Z])')
    charSet += 'A-Z'
  }
  if (config.requireDigit) {
    lookaheads.push('(?=.*\\d)')
    charSet += '\\d'
  }
  if (config.requireSpecialChar) {
    lookaheads.push('(?=.*[!@#$%^&*])')
    charSet += '!@#$%^&*'
  }

  // 如果没有任何要求，匹配任意字符
  if (charSet === '') charSet = '.'
  else charSet = `[${charSet}]`

  return `^${lookaheads.join('')}${charSet}{${config.minLength},}$`
}

5.2 口令策略 (securityManager)

import { securityManager } from '@kit.MDMKit'
import { accountManager } from '@kit.MDMKit'
import { hilog } from '@kit.PerformanceAnalysisKit'
import { Want } from '@kit.AbilityKit'
import { osAccount } from '@kit.BasicServicesKit'

const TAG = 'IdentityService'
const DOMAIN = 0x0003

const ADMIN_WANT: Want = {
  bundleName: 'com.example.securitytool',
  abilityName: 'EnterpriseAdminAbility'
}

export class IdentityService {

  // ========== 口令策略 (securityManager) ==========

  // 设置口令策略
  static setPasswordPolicy(config: PasswordComplexityConfig): boolean {
    try {
      let regex = IdentityService.buildComplexityRegex(config)
      let description = IdentityService.buildDescription(config)
      let policy: securityManager.PasswordPolicy = {
        complexityRegex: regex,
        validityPeriod: 0,   // 有效期由 DomainAccountPolicy 单独管理
        additionalDescription: description
      }
      securityManager.setPasswordPolicy(ADMIN_WANT, policy)
      hilog.info(DOMAIN, TAG, `setPasswordPolicy: regex=${regex}`)
      return true
    } catch (err) {
      hilog.error(DOMAIN, TAG, `setPasswordPolicy failed: ${JSON.stringify(err)}`)
      return false
    }
  }

  // 获取口令策略
  static getPasswordPolicy(): securityManager.PasswordPolicy | null {
    try {
      return securityManager.getPasswordPolicy(ADMIN_WANT)
    } catch (err) {
      hilog.error(DOMAIN, TAG, `getPasswordPolicy failed: ${JSON.stringify(err)}`)
      return null
    }
  }

5.3 域账号策略 (accountManager)

  // ========== 域账号策略 (accountManager) ==========

  // 设置域账号策略（全局）
  static setDomainAccountPolicy(config: DomainPolicyConfig): boolean {
    try {
      let policy: accountManager.DomainAccountPolicy = {
        authenticationValidityPeriod:
          config.authTokenValidityMinutes === -1 ? -1 : config.authTokenValidityMinutes * 60,
        passwordValidityPeriod:
          config.passwordMaxAgeDays === -1 ? -1 : config.passwordMaxAgeDays * 86400,
        passwordExpirationNotification:
          config.expirationNotifyDays * 86400
      }
      // 全局策略: domainAccountInfo 内部属性均为空
      let emptyInfo: osAccount.DomainAccountInfo = {
        domain: '',
        accountName: '',
        serverConfigId: ''
      }
      accountManager.setDomainAccountPolicy(ADMIN_WANT, emptyInfo, policy)
      hilog.info(DOMAIN, TAG, `setDomainAccountPolicy success`)
      return true
    } catch (err) {
      hilog.error(DOMAIN, TAG, `setDomainAccountPolicy failed: ${JSON.stringify(err)}`)
      return false
    }
  }

  // 获取域账号策略（全局）
  static getDomainAccountPolicy(): accountManager.DomainAccountPolicy | null {
    try {
      let emptyInfo: osAccount.DomainAccountInfo = {
        domain: '', accountName: '', serverConfigId: ''
      }
      return accountManager.getDomainAccountPolicy(ADMIN_WANT, emptyInfo)
    } catch (err) {
      hilog.error(DOMAIN, TAG, `getDomainAccountPolicy failed: ${JSON.stringify(err)}`)
      return null
    }
  }
}

7.4 按钮逻辑

// 保存配置
saveConfig() {
  // 1. 收集口令复杂度配置，调用 IdentityService.setPasswordPolicy
  let complexityConfig: PasswordComplexityConfig = {
    minLength: PASSWORD_LENGTH_OPTIONS[this.minLengthIndex],
    requireUppercase: this.requireUppercase,
    requireLowercase: this.requireLowercase,
    requireDigit: this.requireDigit,
    requireSpecialChar: this.requireSpecialChar
  }
  let pwdSuccess = IdentityService.setPasswordPolicy(complexityConfig)

  // 2. 收集域账号策略配置，调用 IdentityService.setDomainAccountPolicy
  let domainConfig: DomainPolicyConfig = {
    passwordMaxAgeDays: PASSWORD_MAX_AGE_OPTIONS[this.maxAgeIndex],
    expirationNotifyDays: EXPIRY_NOTIFY_OPTIONS[this.notifyDaysIndex],
    authTokenValidityMinutes: AUTH_TOKEN_VALIDITY_OPTIONS[this.authValidityIndex]
  }
  let domainSuccess = IdentityService.setDomainAccountPolicy(domainConfig)

  // 3. Toast 反馈
  if (pwdSuccess && domainSuccess) {
    promptAction.showToast({ message: '身份鉴别配置已保存' })
  } else {
    promptAction.showToast({ message: '保存失败，请检查权限' })
  }
}

// 重置默认
resetToDefault() {
  this.minLengthIndex = 1      // 8 位
  this.requireUppercase = true
  this.requireLowercase = true
  this.requireDigit = true
  this.requireSpecialChar = true
  this.maxAgeIndex = 2         // 90 天
  this.notifyDaysIndex = 1     // 7 天
  this.authValidityIndex = 4   // 永久
}

11. 与外设管理模块的架构差异

维度外设管理模块身份鉴别模块
页面结构 4 个 Tab（Tabs 组件） 单页纵向滚动（Scroll + Column）
交互模式 列表 CRUD（增删查）+ 表格 + 弹窗 表单配置（Select + Toggle）+ 保存按钮
弹窗组件 AddWhitelistDialog、AddBlacklistDialog 无弹窗
API 调用模式 即时生效（切换即调用 API） 批量保存（点击"保存配置"统一调用）
数据特点 列表数据（动态增删） 配置数据（固定字段、静态表单）
新增权限 无需新增 需新增 2 个权限 + 重签
核心难点 MDMKit 多组 API 联动 正则表达式生成器、单位换算（天/分钟→秒）
!
API 注意事项：
1. securityManager.setPasswordPolicy 和 accountManager.setDomainAccountPolicy 均为企业级 API，需设备处于 MDM 管理模式。
2. DomainAccountPolicy 中的 passwordValidityPeriod 设置后，用户须在设备侧修改一次域账号密码，否则配置不生效。
3. PasswordPolicy.complexityRegex 仅在用户修改密码时校验，不会影响已有密码。
4. 普通测试设备上 API 调用可能返回 9200001/9200002 权限错误，建议 Service 层做友好提示。
